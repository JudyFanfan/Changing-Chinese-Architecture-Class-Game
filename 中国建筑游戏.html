        // 初始化配对游戏

        function initMatchingGame() {

            const items = document.querySelectorAll('.item');

            const descriptions = document.querySelectorAll('.description');

            

            // 设置拖动事件

            items.forEach(item => {

                item.addEventListener('dragstart', function(e) {

                    e.dataTransfer.setData('text/plain', this.getAttribute('data-id'));

                    this.classList.add('dragging');

                });

                

                item.addEventListener('dragend', function() {

                    this.classList.remove('dragging');

                });

            });

            

            descriptions.forEach(description => {

                description.addEventListener('dragover', function(e) {

                    e.preventDefault();

                    this.classList.add('highlight');

                });

                

                description.addEventListener('dragleave', function() {

                    this.classList.remove('highlight');

                });

                

                description.addEventListener('drop', function(e) {

                    e.preventDefault();

                    this.classList.remove('highlight');

                    

                    const itemId = e.dataTransfer.getData('text/plain');

                    const descriptionId = this.getAttribute('data-id');

                    

                    if(itemId === descriptionId) {

                        // 匹配正确

                        const item = document.querySelector(`.item[data-id="${itemId}"]`);

                        item.style.backgroundColor = '#90EE90';

                        item.style.cursor = 'default';

                        item.draggable = false;

                        

                        this.style.backgroundColor = '#90EE90';

                        this.style.cursor = 'default';

                        

                        matchedPairs++;

                        

                        if(matchedPairs === 4) {

                            document.getElementById('feedback-1').textContent = '恭喜！所有配对都正确！';

                            document.getElementById('feedback-1').classList.add('correct');

                            document.getElementById('next-1').style.display = 'block';

                            createFireworks();

                        }

                    } else {

                        // 匹配错误

                        document.getElementById('feedback-1').textContent = '配对错误，请再试一次！';

                        document.getElementById('feedback-1').classList.add('incorrect');

                        setTimeout(() => {

                            document.getElementById('feedback-1').classList.remove('incorrect');

                        }, 2000);

                    }

                });

            });

        }

        

        // 初始化侦探游戏

        function initDetectiveGame() {

            const options = document.querySelectorAll('#section-2 .option');

            

            options.forEach(option => {

                option.addEventListener('click', function() {

                    // 如果已经回答过，不再处理

                    if(this.classList.contains('correct') || this.classList.contains('incorrect')) {

                        return;

                    }

                    

                    const isCorrect = this.getAttribute('data-correct') === 'true';

                    

                    if(isCorrect) {

                        this.classList.add('correct');

                        correctAnswers++;

                    } else {

                        this.classList.add('incorrect');

                        // 显示正确答案

                        const parent = this.parentElement;

                        const correctOption = parent.querySelector('.option[data-correct="true"]');

                        correctOption.classList.add('correct');

                    }

                    

                    // 禁用所有选项

                    const allOptions = this.parentElement.querySelectorAll('.option');

                    allOptions.forEach(opt => {

                        opt.style.cursor = 'default';

                    });

                    

                    // 检查是否完成所有题目

                    if(correctAnswers === 3) {

                        document.getElementById('feedback-2').textContent = '恭喜！所有问题都回答正确！';

                        document.getElementById('feedback-2').classList.add('correct');

                        document.getElementById('next-2').style.display = 'block';

                        createFireworks();

                    }

                });

            });

        }

        

        // 初始化仿写游戏

        function initWritingGame() {

            document.getElementById('submit-writing').addEventListener('click', function() {

                const text = document.getElementById('writing-area').value;

                

                if(text.length < 100) {

                    document.getElementById('feedback-3').textContent = '您的文字不足100字，请继续补充！';

                    document.getElementById('feedback-3').classList.add('incorrect');

                    return;

                }

                

                // 简单验证：检查是否包含关键词

                const keywords = ['苏州', '园林', '水乡', '传统', '现代', '贝聿铭'];

                let foundKeywords = 0;

                

                keywords.forEach(keyword => {

                    if(text.includes(keyword)) {

                        foundKeywords++;

                    }

                });

                

                if(foundKeywords >= 2) {

                    document.getElementById('feedback-3').textContent = '提交成功！您的描述很好地体现了苏州建筑的特点。';

                    document.getElementById('feedback-3').classList.add('correct');

                    document.getElementById('next-3').style.display = 'block';

                    writingSubmitted = true;

                    createFireworks();

                } else {

                    document.getElementById('feedback-3').textContent = '您的描述可以更多地体现苏州建筑的特色，请再完善一下！';

                    document.getElementById('feedback-3').classList.add('incorrect');

                }

            });

        }

        // 创建礼花效果

        function createFireworks() {

            const fireworksContainer = document.getElementById('fireworks');

            fireworksContainer.style.display = 'block';

            

            // 创建多个礼花

            for(let i = 0; i < 50; i++) {

                const firework = document.createElement('div');

                firework.className = 'firework';

                

                // 随机颜色

                const colors = ['#FF0000', '#FFD700', '#00FF00', '#0000FF', '#FF00FF', '#00FFFF'];

                const color = colors[Math.floor(Math.random() * colors.length)];

                firework.style.backgroundColor = color;

                firework.style.boxShadow = `0 0 10px 2px ${color}`;

                

                // 随机位置

                const startX = Math.random() * window.innerWidth;

                const startY = Math.random() * window.innerHeight;

                

                firework.style.left = `${startX}px`;

                firework.style.top = `${startY}px`;

                

                // 随机爆炸方向

                const tx = (Math.random() - 0.5) * 200;

                const ty = (Math.random() - 0.5) * 200;

                

                firework.style.setProperty('--tx', `${tx}px`);

                firework.style.setProperty('--ty', `${ty}px`);

                

                fireworksContainer.appendChild(firework);

                

                // 移除元素

                setTimeout(() => {

                    firework.remove();

                }, 1000);

            }

            

            // 隐藏礼花容器

            setTimeout(() => {

                fireworksContainer.style.display = 'none';

            }, 1000);

        }

    </script>
